# Test 1
# Testujemy czy funkcja enter() jest wywoływana poprawnie
@manticore.test
def test_enter_function():
    m = manticore.ManticoreEVM()
    lottery = m.solidity_create_contract('Lottery.sol')
    lottery.enter(value=m.make_wei(1, 'ether'))
    assert m.get_balance(lottery.address) == m.make_wei(1, 'ether')
